#include <iostream>
#include <utility>
#include <functional>
#include <memory>
#include <cassert>

// 红黑树颜色
enum class Color { Red, Black };

// 节点结构体（使用 NIL 哨兵）
template<typename Key, typename Value>
struct RBNode {
    std::pair<const Key, Value> data;
    RBNode* left;
    RBNode* right;
    RBNode* parent;
    Color color;

    RBNode(const Key& k, const Value& v, Color c = Color::Red)
        : data(k, v), left(nullptr), right(nullptr), parent(nullptr), color(c) {}
};

// 前向声明
template<typename Key, typename Value, typename Compare>
class Map;

// 迭代器（支持 const 和非 const）
template<typename Key, typename Value, bool IsConst = false>
class MapIterator {
public:
    using iterator_category = std::bidirectional_iterator_tag;
    using value_type = std::pair<const Key, Value>;
    using difference_type = std::ptrdiff_t;
    using pointer = typename std::conditional<IsConst, const value_type*, value_type*>::type;
    using reference = typename std::conditional<IsConst, const value_type&, value_type&>::type;

    MapIterator() : node(nullptr), root(nullptr) {}
    MapIterator(RBNode<Key, Value>* n, RBNode<Key, Value>* r) : node(n), root(r) {}

    reference operator*() const { return node->data; }
    pointer operator->() const { return &node->data; }

    MapIterator& operator++() {
        node = successor(node);
        return *this;
    }
    MapIterator operator++(int) {
        MapIterator tmp = *this;
        ++(*this);
        return tmp;
    }
    MapIterator& operator--() {
        node = predecessor(node);
        return *this;
    }
    MapIterator operator--(int) {
        MapIterator tmp = *this;
        --(*this);
        return tmp;
    }

    bool operator==(const MapIterator& other) const { return node == other.node; }
    bool operator!=(const MapIterator& other) const { return node != other.node; }

private:
    RBNode<Key, Value>* node;
    RBNode<Key, Value>* root;

    // 后继（中序遍历的下一个）
    RBNode<Key, Value>* successor(RBNode<Key, Value>* x) const {
        if (x->right && x->right != root->parent) {  // root->parent 是 NIL
            x = x->right;
            while (x->left && x->left != root->parent) x = x->left;
            return x;
        }
        RBNode<Key, Value>* y = x->parent;
        while (y && y != root->parent && x == y->right) {
            x = y;
            y = y->parent;
        }
        return y;
    }

    // 前驱（中序遍历的前一个）
    RBNode<Key, Value>* predecessor(RBNode<Key, Value>* x) const {
        if (x->left && x->left != root->parent) {
            x = x->left;
            while (x->right && x->right != root->parent) x = x->right;
            return x;
        }
        RBNode<Key, Value>* y = x->parent;
        while (y && y != root->parent && x == y->left) {
            x = y;
            y = y->parent;
        }
        return y;
    }
};

// Map 容器
template<typename Key, typename Value, typename Compare = std::less<Key>>
class Map {
public:
    using key_type = Key;
    using mapped_type = Value;
    using value_type = std::pair<const Key, Value>;
    using size_type = size_t;
    using iterator = MapIterator<Key, Value, false>;
    using const_iterator = MapIterator<Key, Value, true>;

    // 构造和析构
    /**
     *  @brief  Default constructor.
     *
     *  Creates an empty map with no elements.  The sentinel node is
     *  initialized and the root points to it.
     */
    Map() : size_(0) {
        nil = new RBNode<Key, Value>(Key(), Value(), Color::Black);
        nil->left = nil->right = nil->parent = nil;
        root = nil;
    }
    /**
     *  @brief  Copy constructor.
     *  @param  __x  Another map to copy from.
     *
     *  Performs a deep copy of all nodes in @p __x, preserving the
     *  red‑black tree structure and element order.
     */
    Map(const Map& other) : size_(0) {
        nil = new RBNode<Key, Value>(Key(), Value(), Color::Black);
        nil->left = nil->right = nil->parent = nil;
        root = nil;
        // 深拷贝
        copyTree(other.root, nullptr);
    }
    /**
     *  @brief  Move constructor.
     *  @param  __x  Another map to move from.
     *
     *  Transfers ownership of the internal tree from @p __x to the newly
     *  constructed map.  The moved‑from map is left in a valid but
     *  unspecified state.
     */
    Map(Map&& other) noexcept
        : root(other.root), size_(other.size_), nil(other.nil) {
        other.root = nullptr;
        other.nil = nullptr;
        other.size_ = 0;
    }
    /**
     *  @brief  Copy assignment operator.
     *  @param  __x  Another map to copy from.
     *  @return  Reference to this map.
     *
     *  Replaces the contents of this map with a deep copy of @p __x.
     */
    Map& operator=(const Map& other) {
        if (this != &other) {
            clear();
            delete nil;
            nil = new RBNode<Key, Value>(Key(), Value(), Color::Black);
            nil->left = nil->right = nil->parent = nil;
            root = nil;
            copyTree(other.root, nullptr);
        }
        return *this;
    }
    /**
     *  @brief  Move assignment operator.
     *  @param  __x  Another map to move from.
     *  @return  Reference to this map.
     *
     *  Transfers the internal tree from @p __x to this map.
     */
    Map& operator=(Map&& other) noexcept {
        if (this != &other) {
            clear();
            delete nil;
            root = other.root;
            nil = other.nil;
            size_ = other.size_;
            other.root = nullptr;
            other.nil = nullptr;
            other.size_ = 0;
        }
        return *this;
    }
    /**
     *  @brief  Destructor.
     *
     *  Deallocates all nodes and the sentinel node, releasing all
     *  resources held by the map.
     */
    ~Map() {
        clear();
        delete nil;
    }

    // 迭代器
    /**
     *  @brief  Returns an iterator referring to the first element.
     *  @return  An iterator to the smallest key.
     *
     *  The iterator can be used to traverse the map in ascending order.
     */
    iterator begin() {
        return iterator(minimum(root), root);
    }
    /**
     *  @brief  Returns a const iterator referring to the first element.
     *  @return  A const iterator to the smallest key.
     */
    const_iterator begin() const {
        return const_iterator(minimum(root), root);
    }
    /**
     *  @brief  Returns a const iterator referring to the first element.
     *  @return  A const iterator to the smallest key.
     */
    const_iterator cbegin() const { return begin(); }
    /**
     *  @brief  Returns an iterator referring to the past‑the‑end element.
     *  @return  An iterator that denotes the end of the container.
     */
    iterator end() {
        return iterator(nil, root);
    }
    /**
     *  @brief  Returns a const iterator referring to the past‑the‑end element.
     *  @return  A const iterator that denotes the end of the container.
     */
    const_iterator end() const {
        return const_iterator(nil, root);
    }
    /**
     *  @brief  Returns a const iterator referring to the past‑the‑end element.
     *  @return  A const iterator that denotes the end of the container.
     */
    const_iterator cend() const { return end(); }

    // 容量
    /**
     *  @brief  Checks whether the container is empty.
     *  @return  True if the container contains no elements, false otherwise.
     */
    bool empty() const { return size_ == 0; }
    /**
     *  @brief  Returns the number of elements in the container.
     *  @return  The number of stored key‑value pairs.
     */
    size_type size() const { return size_; }

    // 元素访问
    /**
     *  @brief  Access element with the given key, inserting a default‑constructed
     *          value if the key does not exist.
     *  @param  __key  The key of the element to find or insert.
     *  @return  Reference to the mapped value associated with @p __key.
     *
     *  If @p __key matches an element, it returns a reference to its mapped value.
     *  Otherwise, the function inserts a new element with that key and a
     *  value‑initialized mapped type, and returns a reference to it.
     */
    Value& operator[](const Key& key) {
        iterator it = find(key);
        if (it != end())
            return it->second;
        auto ret = insert(key, Value());
        return ret.first->second;
    }

    // 查找
    /**
     *  @brief  Finds an element with a given key.
     *  @param  __key  The key to search for.
     *  @return  An iterator to the element if found, or end() otherwise.
     */
    iterator find(const Key& key) {
        RBNode<Key, Value>* cur = root;
        while (cur != nil) {
            if (comp(key, cur->data.first))
                cur = cur->left;
            else if (comp(cur->data.first, key))
                cur = cur->right;
            else
                return iterator(cur, root);
        }
        return end();
    }
    /**
     *  @brief  Finds an element with a given key (const version).
     *  @param  __key  The key to search for.
     *  @return  A const iterator to the element if found, or end() otherwise.
     */
    const_iterator find(const Key& key) const {
        RBNode<Key, Value>* cur = root;
        while (cur != nil) {
            if (comp(key, cur->data.first))
                cur = cur->left;
            else if (comp(cur->data.first, key))
                cur = cur->right;
            else
                return const_iterator(cur, root);
        }
        return end();
    }

    // 计数（0 或 1）
    /**
     *  @brief  Counts the number of elements with a given key.
     *  @param  __key  The key to count.
     *  @return  One if the key exists, zero otherwise.
     *
     *  Because the map does not allow duplicate keys, the result is either 0 or 1.
     */
    size_type count(const Key& key) const {
        return find(key) != end() ? 1 : 0;
    }

    // 插入
    /**
     *  @brief  Inserts a key‑value pair.
     *  @param  __value  The value to insert (a pair of key and mapped value).
     *  @return  A pair consisting of an iterator to the inserted element
     *          (or the element that prevented insertion) and a bool indicating
     *          whether the insertion took place.
     *
     *  If the key already exists, the insertion is not performed and the
     *  returned iterator points to the existing element.
     */
    std::pair<iterator, bool> insert(const value_type& value) {
        return insert(value.first, value.second);
    }

    /**
     *  @brief  Inserts a key‑value pair (overload).
     *  @param  __key    The key to insert.
     *  @param  __value  The value associated with the key.
     *  @return  A pair consisting of an iterator to the inserted element
     *          (or the element that prevented insertion) and a bool indicating
     *          whether the insertion took place.
     *
     *  This is a convenience overload that constructs a pair internally.
     */
    std::pair<iterator, bool> insert(const Key& key, const Value& val) {
        RBNode<Key, Value>* z = new RBNode<Key, Value>(key, val);
        z->left = z->right = nil;
        RBNode<Key, Value>* y = nil;
        RBNode<Key, Value>* x = root;
        while (x != nil) {
            y = x;
            if (comp(z->data.first, x->data.first))
                x = x->left;
            else if (comp(x->data.first, z->data.first))
                x = x->right;
            else {
                delete z;
                return {iterator(x, root), false};
            }
        }
        z->parent = y;
        if (y == nil)
            root = z;
        else if (comp(z->data.first, y->data.first))
            y->left = z;
        else
            y->right = z;
        insertFixup(z);
        ++size_;
        return {iterator(z, root), true};
    }

    // 删除
    /**
     *  @brief  Erases an element at a given position.
     *  @param  __pos  An iterator pointing to the element to erase.
     *
     *  The iterator becomes invalid after the operation.  The red‑black tree
     *  properties are maintained.
     */
    void erase(iterator pos) {
        if (pos == end()) return;
        RBNode<Key, Value>* z = pos.node;
        RBNode<Key, Value>* y = z;
        RBNode<Key, Value>* x = nullptr;
        Color y_original_color = y->color;
        if (z->left == nil) {
            x = z->right;
            transplant(z, z->right);
        } else if (z->right == nil) {
            x = z->left;
            transplant(z, z->left);
        } else {
            y = minimum(z->right);
            y_original_color = y->color;
            x = y->right;
            if (y->parent == z) {
                if (x != nil) x->parent = y;
            } else {
                transplant(y, y->right);
                y->right = z->right;
                y->right->parent = y;
            }
            transplant(z, y);
            y->left = z->left;
            y->left->parent = y;
            y->color = z->color;
        }
        delete z;
        --size_;
        if (y_original_color == Color::Black)
            deleteFixup(x);
    }

    /**
     *  @brief  Erases an element with a given key.
     *  @param  __key  The key of the element to erase.
     *  @return  One if the element was removed, zero otherwise.
     */
    size_type erase(const Key& key) {
        iterator it = find(key);
        if (it == end()) return 0;
        erase(it);
        return 1;
    }

    // 清空
    /**
     *  @brief  Removes all elements from the container.
     *
     *  The container becomes empty.  All iterators and references are
     *  invalidated.
     */
    void clear() {
        clear(root);
        root = nil;
        size_ = 0;
    }

private:
    RBNode<Key, Value>* root;
    RBNode<Key, Value>* nil;   // 哨兵节点，黑色，所有空指针指向它
    size_type size_;
    Compare comp;

    // 左旋
    void leftRotate(RBNode<Key, Value>* x) {
        RBNode<Key, Value>* y = x->right;
        x->right = y->left;
        if (y->left != nil) y->left->parent = x;
        y->parent = x->parent;
        if (x->parent == nil)
            root = y;
        else if (x == x->parent->left)
            x->parent->left = y;
        else
            x->parent->right = y;
        y->left = x;
        x->parent = y;
    }

    // 右旋
    void rightRotate(RBNode<Key, Value>* y) {
        RBNode<Key, Value>* x = y->left;
        y->left = x->right;
        if (x->right != nil) x->right->parent = y;
        x->parent = y->parent;
        if (y->parent == nil)
            root = x;
        else if (y == y->parent->left)
            y->parent->left = x;
        else
            y->parent->right = x;
        x->right = y;
        y->parent = x;
    }

    // 插入修复
    void insertFixup(RBNode<Key, Value>* z) {
        while (z->parent->color == Color::Red) {
            if (z->parent == z->parent->parent->left) {
                RBNode<Key, Value>* y = z->parent->parent->right;
                if (y->color == Color::Red) {
                    z->parent->color = Color::Black;
                    y->color = Color::Black;
                    z->parent->parent->color = Color::Red;
                    z = z->parent->parent;
                } else {
                    if (z == z->parent->right) {
                        z = z->parent;
                        leftRotate(z);
                    }
                    z->parent->color = Color::Black;
                    z->parent->parent->color = Color::Red;
                    rightRotate(z->parent->parent);
                }
            } else {
                RBNode<Key, Value>* y = z->parent->parent->left;
                if (y->color == Color::Red) {
                    z->parent->color = Color::Black;
                    y->color = Color::Black;
                    z->parent->parent->color = Color::Red;
                    z = z->parent->parent;
                } else {
                    if (z == z->parent->left) {
                        z = z->parent;
                        rightRotate(z);
                    }
                    z->parent->color = Color::Black;
                    z->parent->parent->color = Color::Red;
                    leftRotate(z->parent->parent);
                }
            }
        }
        root->color = Color::Black;
    }

    // 删除修复（x 可能是 nil）
    void deleteFixup(RBNode<Key, Value>* x) {
        while (x != root && x->color == Color::Black) {
            if (x == x->parent->left) {
                RBNode<Key, Value>* w = x->parent->right;
                if (w->color == Color::Red) {
                    w->color = Color::Black;
                    x->parent->color = Color::Red;
                    leftRotate(x->parent);
                    w = x->parent->right;
                }
                if (w->left->color == Color::Black && w->right->color == Color::Black) {
                    w->color = Color::Red;
                    x = x->parent;
                } else {
                    if (w->right->color == Color::Black) {
                        w->left->color = Color::Black;
                        w->color = Color::Red;
                        rightRotate(w);
                        w = x->parent->right;
                    }
                    w->color = x->parent->color;
                    x->parent->color = Color::Black;
                    w->right->color = Color::Black;
                    leftRotate(x->parent);
                    x = root;
                }
            } else {
                RBNode<Key, Value>* w = x->parent->left;
                if (w->color == Color::Red) {
                    w->color = Color::Black;
                    x->parent->color = Color::Red;
                    rightRotate(x->parent);
                    w = x->parent->left;
                }
                if (w->right->color == Color::Black && w->left->color == Color::Black) {
                    w->color = Color::Red;
                    x = x->parent;
                } else {
                    if (w->left->color == Color::Black) {
                        w->right->color = Color::Black;
                        w->color = Color::Red;
                        leftRotate(w);
                        w = x->parent->left;
                    }
                    w->color = x->parent->color;
                    x->parent->color = Color::Black;
                    w->left->color = Color::Black;
                    rightRotate(x->parent);
                    x = root;
                }
            }
        }
        x->color = Color::Black;
    }

    // 移植子树
    void transplant(RBNode<Key, Value>* u, RBNode<Key, Value>* v) {
        if (u->parent == nil)
            root = v;
        else if (u == u->parent->left)
            u->parent->left = v;
        else
            u->parent->right = v;
        v->parent = u->parent;
    }

    // 最小节点
    RBNode<Key, Value>* minimum(RBNode<Key, Value>* x) const {
        while (x->left != nil) x = x->left;
        return x;
    }

    // 递归清空子树（不包括 nil）
    void clear(RBNode<Key, Value>* node) {
        if (node == nil) return;
        clear(node->left);
        clear(node->right);
        delete node;
    }

    // 深拷贝辅助函数
    void copyTree(RBNode<Key, Value>* src, RBNode<Key, Value>* parent) {
        if (src == nil) return;
        RBNode<Key, Value>* newNode = new RBNode<Key, Value>(src->data.first, src->data.second, src->color);
        newNode->parent = parent;
        newNode->left = nil;
        newNode->right = nil;
        if (parent == nil)
            root = newNode;
        else if (comp(newNode->data.first, parent->data.first))
            parent->left = newNode;
        else
            parent->right = newNode;
        copyTree(src->left, newNode);
        copyTree(src->right, newNode);
    }
};

// 简单测试
int main() {
    Map<int, std::string> m;
    m.insert({1, "one"});
    m[2] = "two";
    m[3] = "three";
    m[4] = "four";

    for (auto it = m.begin(); it != m.end(); ++it) {
        std::cout << it->first << " -> " << it->second << std::endl;
    }

    auto it = m.find(2);
    if (it != m.end()) {
        std::cout << "Found: " << it->second << std::endl;
        m.erase(it);
    }

    std::cout << "After erase 2:\n";
    for (const auto& p : m) {
        std::cout << p.first << " -> " << p.second << std::endl;
    }

    m.erase(3);
    std::cout << "After erase 3, size = " << m.size() << std::endl;

    Map<int, std::string> m2 = m;  // 拷贝构造
    std::cout << "Copied map:\n";
    for (const auto& p : m2) {
        std::cout << p.first << " -> " << p.second << std::endl;
    }
    return 0;
}
