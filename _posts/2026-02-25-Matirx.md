```cpp
class Node
{
    private:
    U n,m;

    public:
    Node(U x,U y)
    {
        n=x;
        m=y;
        memset(mp,0,sizeof(mp));

            n = m = 0;

        n = v.size();
        m = v[0].size();
        memset(mp, 0, sizeof(mp));
        for(U i=0; i<n and i+1<N; i++)
        {
            for(U j=0; j<m and j+1<N; j++)
            {
                mp[i+1][j+1] = v[i][j];
            }
        }

    {
        n = other.n;

            mp[i][i]=1;
        return ;
    }

    {
        if(this != &other)
        {
            n = other.n;
            m = other.m;
            memcpy(mp, other.mp, sizeof(mp));
        }
        return *this;
    }

    {
        if(m != x.n)
        {
            cerr << "Matrix dimensions do not match for multiplication!" << '\n';
            return Node();
        }
        
        T p=x.m;
        Node res(n,p);
        for(T i=1;i<=n;i++)
        {
            for(T j=1;j<=p;j++)
            {
                long long sum = 0; 
                for(T k=1;k<=m;k++)
                {
                    sum += (1LL * mp[i][k] * x.mp[k][j]) % mod;
                    if(sum >= mod) sum -= mod;
                }
                res.mp[i][j] = sum % mod;
            }
        }
        return res;
    }

    {
        if(n != m)
        {
            cerr << "Cannot exponentiate non-square matrix!" << '\n';
            return;
        }
        Node result(n);
        result.unit();
        Node base = *this;
        long long exp = k;
        while(exp > 0)
        {
            if(exp & 1) result = result * base;
            base = base * base;
            exp >>= 1;
        }
        *this = result;
    }

    {
        if(x>n or y>n)

            cerr<< "The type of Matrix is not same as the type of data. Chech your data structer.";

        }
        mp[x-1][y-1]=data;
        return ;
    }

        if(n != x.n || m != x.m)
        {
            cerr << "Matrix dimensions do not match for addition!" << '\n';
            return Node();
        }
        
        Node res(n,m);
        for(T i=1;i<=n;i++)
        {
            for(T j=1;j<=m;j++)
            {
                res.mp[i][j] = (mp[i][j] + x.mp[i][j]) % mod;
            }

            for(T j=1;j<=m;j++)
            {
                res.mp[i][j] = (mp[i][j] - x.mp[i][j] + mod) % mod;
            }
        }
        return res;

        if(n!=x.n || m!=x.m) return false;
```

```cpp
#include <iostream>
#include <cstring>
#include <vector>
#include <stdexcept>
using namespace std;

/**
 * @class Node
 * @brief Dynamic matrix class supporting basic matrix operations and bitwise operations.
 * @ingroup matrix
 *
 * This class encapsulates a dynamically allocated 2D matrix stored in row-major order.
 * It provides construction, copy control, arithmetic operations (addition, subtraction,
 * multiplication, exponentiation) and matrix-like bitwise operations (AND, OR, XOR).
 * Arithmetic results are taken modulo a user-specified modulus passed as a function parameter,
 * while bitwise operations directly accumulate the bitwise results (no modulo).
 * Indices start from 1, following mathematical convention.
 *
 * When dimension mismatches or illegal operations (e.g., exponentiation of a non-square matrix)
 * are detected, an std::invalid_argument exception is thrown.
 */
class Node
{
private:
    size_t n, m;          //!< Number of rows (n) and columns (m)
    size_t* data;         //!< Dynamically allocated 1D array storing all elements in row-major order, size n * m

public:
    /**
     * @brief Constructor with specified rows and columns, elements zero-initialized.
     * @param rows Number of rows.
     * @param cols Number of columns.
     *
     * Allocates a dynamic array of size rows * cols and value-initializes all elements to zero.
     * If rows or cols is zero, an empty matrix (data = nullptr) is created.
     */
    Node(size_t rows, size_t cols) : n(rows), m(cols)
    {
        if (n == 0 || m == 0)
        {
            n = m = 0;
            data = nullptr;
            return;
        }
        data = new size_t[n * m]();  // value-initialization to zero
    }

    /**
     * @brief Default constructor, creates an empty matrix (0x0).
     */
    Node() : n(0), m(0), data(nullptr) {}

    /**
     * @brief Constructor creating an n x n square matrix, elements zero-initialized.
     * @param dim Dimension of the square matrix.
     */
    explicit Node(size_t dim) : Node(dim, dim) {}

    /**
     * @brief Construct a matrix from a 2D vector.
     * @param v 2D vector providing the data.
     *
     * The number of rows is v.size(), and the number of columns is v[0].size().
     * If the vector is empty, an empty matrix is created.
     * Data is copied from the vector into the internal dynamic array; indices start from 1.
     */
    Node(const vector<vector<size_t>>& v)
    {
        if (v.empty())
        {
            n = m = 0;
            data = nullptr;
            return;
        }
        n = v.size();
        m = v[0].size();
        data = new size_t[n * m]();
        for (size_t i = 0; i < n; ++i)
        {
            for (size_t j = 0; j < m; ++j)
            {
                // Internal index: (i * m + j) corresponds to mathematical coordinate (i+1, j+1)
                data[i * m + j] = v[i][j];
            }
        }
    }

    /**
     * @brief Copy constructor.
     * @param other Matrix to copy from.
     */
    Node(const Node& other) : n(other.n), m(other.m)
    {
        if (other.data)
        {
            data = new size_t[n * m];
            memcpy(data, other.data, n * m * sizeof(size_t));
        }
        else
        {
            data = nullptr;
        }
    }

    /**
     * @brief Move constructor.
     * @param other Matrix to move from (left in empty state).
     */
    Node(Node&& other) noexcept : n(other.n), m(other.m), data(other.data)
    {
        other.n = other.m = 0;
        other.data = nullptr;
    }

    /**
     * @brief Destructor, releases dynamic memory.
     */
    ~Node()
    {
        delete[] data;
    }

    /**
     * @brief Copy assignment operator.
     * @param other Matrix to assign from.
     * @return Reference to this object.
     */
    Node& operator=(const Node& other)
    {
        if (this != &other)
        {
            delete[] data;  // Free existing memory

            n = other.n;
            m = other.m;
            if (other.data)
            {
                data = new size_t[n * m];
                memcpy(data, other.data, n * m * sizeof(size_t));
            }
            else
            {
                data = nullptr;
            }
        }
        return *this;
    }

    /**
     * @brief Move assignment operator.
     * @param other Matrix to move from.
     * @return Reference to this object.
     */
    Node& operator=(Node&& other) noexcept
    {
        if (this != &other)
        {
            delete[] data;
            n = other.n;
            m = other.m;
            data = other.data;
            other.n = other.m = 0;
            other.data = nullptr;
        }
        return *this;
    }

    /**
     * @brief Access element at (x, y) (indices starting from 1).
     * @param x Row index (1 ≤ x ≤ n).
     * @param y Column index (1 ≤ y ≤ m).
     * @return Value at (x, y).
     *
     * @note No bounds checking is performed; the caller must ensure indices are valid.
     */
    size_t query(size_t x, size_t y) const
    {
        // Row-major conversion: (x,y) maps to internal index (x-1)*m + (y-1)
        return data[(x - 1) * m + (y - 1)];
    }

    /**
     * @brief Initialize the matrix as the identity matrix (only for square matrices).
     *
     * @throw std::invalid_argument if the matrix is not square.
     */
    void unit()
    {
        if (n != m)
        {
            throw invalid_argument("unit() called on non-square matrix");
        }
        // Zero out all elements
        memset(data, 0, n * m * sizeof(size_t));
        // Set diagonal elements to 1
        for (size_t i = 0; i < n; ++i)
        {
            data[i * m + i] = 1;  // internal index (i,i) corresponds to (i+1,i+1)
        }
    }

    /**
     * @brief Read matrix elements from standard input.
     *
     * Reads n * m integers in row-major order, indices starting from (1,1).
     */
    void input()
    {
        for (size_t i = 0; i < n; ++i)
        {
            for (size_t j = 0; j < m; ++j)
            {
                cin >> data[i * m + j];
            }
        }
    }

    /**
     * @brief Print the matrix to standard output.
     *
     * Outputs elements in row-major order, separated by spaces, with a newline after each row.
     * Note: Prints the raw stored values, not modulo-adjusted.
     */
    void print() const
    {
        for (size_t i = 0; i < n; ++i)
        {
            for (size_t j = 0; j < m; ++j)
            {
                cout << data[i * m + j];
                if (j < m - 1) cout << ' ';
            }
            cout << '\n';
        }
    }

    /**
     * @brief Matrix addition.
     * @param other Matrix to add.
     * @param mod  Modulus for the result.
     * @return A new matrix representing *this + other, modulo mod.
     *
     * @throw std::invalid_argument if the dimensions of the two matrices do not match.
     * Each element of the result is (this[i][j] + other[i][j]) % mod.
     */
    Node add(const Node& other, long long mod) const
    {
        if (n != other.n || m != other.m)
        {
            throw invalid_argument("Matrix dimensions do not match for addition");
        }
        Node res(n, m);
        for (size_t i = 0; i < n * m; ++i)
        {
            res.data[i] = (data[i] + other.data[i]) % mod;
        }
        return res;
    }

    /**
     * @brief Matrix subtraction.
     * @param other Matrix to subtract.
     * @param mod  Modulus for the result.
     * @return A new matrix representing *this - other, modulo mod.
     *
     * @throw std::invalid_argument if the dimensions of the two matrices do not match.
     * Each element of the result is (this[i][j] - other[i][j] + mod) % mod.
     */
    Node subtract(const Node& other, long long mod) const
    {
        if (n != other.n || m != other.m)
        {
            throw invalid_argument("Matrix dimensions do not match for subtraction");
        }
        Node res(n, m);
        for (size_t i = 0; i < n * m; ++i)
        {
            long long diff = (static_cast<long long>(data[i]) - other.data[i]) % mod;
            if (diff < 0) diff += mod;
            res.data[i] = static_cast<size_t>(diff);
        }
        return res;
    }

    /**
     * @brief Matrix multiplication.
     * @param other Right-hand side matrix.
     * @param mod  Modulus for the result.
     * @return A new matrix representing *this * other, modulo mod.
     *
     * @throw std::invalid_argument if the number of columns of the left matrix does not equal
     *        the number of rows of the right matrix.
     * The result matrix has dimensions n x other.m.
     * Each element is Σ (this[i][k] * other[k][j]) % mod.
     */
    Node multiply(const Node& other, long long mod) const
    {
        if (m != other.n)
        {
            throw invalid_argument("Matrix dimensions do not match for multiplication");
        }

        Node res(n, other.m);
        for (size_t i = 0; i < n; ++i)
        {
            for (size_t k = 0; k < m; ++k)
            {
                size_t aik = data[i * m + k];
                if (aik == 0) continue;  // simple optimization
                for (size_t j = 0; j < other.m; ++j)
                {
                    size_t bkj = other.data[k * other.m + j];
                    // Accumulate product using long long to avoid overflow
                    res.data[i * res.m + j] = (res.data[i * res.m + j] +
                                               (static_cast<long long>(aik) * bkj) % mod) % mod;
                }
            }
        }
        return res;
    }

    /**
     * @brief Matrix exponentiation (only for square matrices).
     * @param exponent Non-negative exponent.
     * @param mod      Modulus for the result.
     * @return A new matrix representing this^exponent, modulo mod.
     *
     * @throw std::invalid_argument if the matrix is not square.
     * Uses fast exponentiation (binary exponentiation).
     */
    Node power(long long exponent, long long mod) const
    {
        if (n != m)
        {
            throw invalid_argument("Cannot exponentiate non-square matrix");
        }
        if (exponent == 0)
        {
            Node res(n);   // n x n zero matrix
            res.unit();    // becomes identity matrix
            return res;
        }

        Node result(n);
        result.unit();
        Node base = *this;
        long long exp = exponent;
        while (exp > 0)
        {
            if (exp & 1)
                result = result.multiply(base, mod);
            base = base.multiply(base, mod);
            exp >>= 1;
        }
        return result;
    }

    /**
     * @brief Bitwise XOR operation (matrix multiplication-like but using XOR as inner product).
     * @param other Right-hand side matrix.
     * @return A new matrix where each element = Σ (this[i][k] ^ other[k][j]) (no modulo).
     *
     * @throw std::invalid_argument if the number of columns of the left matrix does not equal
     *        the number of rows of the right matrix.
     */
    Node bitwiseXor(const Node& other) const
    {
        if (m != other.n)
        {
            throw invalid_argument("Matrix dimensions do not match for bitwise XOR");
        }

        Node res(n, other.m);
        for (size_t i = 0; i < n; ++i)
        {
            for (size_t k = 0; k < m; ++k)
            {
                size_t aik = data[i * m + k];
                for (size_t j = 0; j < other.m; ++j)
                {
                    res.data[i * res.m + j] += aik ^ other.data[k * other.m + j];
                }
            }
        }
        return res;
    }

    /**
     * @brief Bitwise AND operation (matrix multiplication-like but using AND as inner product).
     * @param other Right-hand side matrix.
     * @return A new matrix where each element = Σ (this[i][k] & other[k][j]).
     *
     * @throw std::invalid_argument if the number of columns of the left matrix does not equal
     *        the number of rows of the right matrix.
     */
    Node bitwiseAnd(const Node& other) const
    {
        if (m != other.n)
        {
            throw invalid_argument("Matrix dimensions do not match for bitwise AND");
        }

        Node res(n, other.m);
        for (size_t i = 0; i < n; ++i)
        {
            for (size_t k = 0; k < m; ++k)
            {
                size_t aik = data[i * m + k];
                for (size_t j = 0; j < other.m; ++j)
                {
                    res.data[i * res.m + j] += aik & other.data[k * other.m + j];
                }
            }
        }
        return res;
    }

    /**
     * @brief Bitwise OR operation (matrix multiplication-like but using OR as inner product).
     * @param other Right-hand side matrix.
     * @return A new matrix where each element = Σ (this[i][k] | other[k][j]).
     *
     * @throw std::invalid_argument if the number of columns of the left matrix does not equal
     *        the number of rows of the right matrix.
     */
    Node bitwiseOr(const Node& other) const
    {
        if (m != other.n)
        {
            throw invalid_argument("Matrix dimensions do not match for bitwise OR");
        }

        Node res(n, other.m);
        for (size_t i = 0; i < n; ++i)
        {
            for (size_t k = 0; k < m; ++k)
            {
                size_t aik = data[i * m + k];
                for (size_t j = 0; j < other.m; ++j)
                {
                    res.data[i * res.m + j] += aik | other.data[k * other.m + j];
                }
            }
        }
        return res;
    }

    /**
     * @brief Equality comparison operator.
     * @param other Matrix to compare with.
     * @return true if both matrices have the same dimensions and identical elements, false otherwise.
     */
    bool operator==(const Node& other) const
    {
        if (n != other.n || m != other.m) return false;
        return memcmp(data, other.data, n * m * sizeof(size_t)) == 0;
    }

    /**
     * @brief Get number of rows.
     */
    size_t rows() const { return n; }

    /**
     * @brief Get number of columns.
     */
    size_t cols() const { return m; }
};
```
