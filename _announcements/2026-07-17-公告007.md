```cpp
/**
 * main.cpp
 * 
 * 将 LaTeX algorithmic 伪代码转换为 Pseudocode.js 可识别的格式。
 * 用法: ./main <输入文件> [输出文件]
 * 
 * 功能：
 *   - 自动提取所有 \begin{algorithmic}...\end{algorithmic} 块
 *   - 去除外层 \begin{algorithm}...\end{algorithm}
 *   - 去除 \begin{algorithmic} 的可选参数（如 [1]）
 *   - 修复拼写错误：\CommentRR → \Comment
 *   - 删除无效命令：\quad
 *   - 转换 \Call{func}{arg} → func(arg)（避免数学模式冲突）
 *   - 移除 \_ 转义（\_right → _right）
 *   - 替换 \text{null} → null
 *   - 移除控制结构前的 \State（\State \If → \If）
 *   - 将 \State \Return 统一转换为 \Return
 *   - 合并多余空行
 */

#include <iostream>
#include <fstream>
#include <string>
#include <regex>
#include <sstream>

/**
 * 转换单个 algorithmic 内部的算法体
 */
std::string convertAlgorithmicContent(const std::string& input) {
    std::string result = input;

    // 1. 修复拼写错误 \CommentRR -> \Comment
    std::regex commentRR(R"(\\CommentRR)");
    result = std::regex_replace(result, commentRR, "\\Comment");

    // 2. 删除所有 \quad（无效命令）
    std::regex quad(R"(\\quad\s*)");
    result = std::regex_replace(result, quad, "");

    // 3. 将 \Call{func}{arg} 转换为 func(arg)
    //    注意：此正则假定参数不包含嵌套花括号
    std::regex callPattern(R"(\\Call\{([^}]*)\}\{([^}]*)\})");
    result = std::regex_replace(result, callPattern, "$1($2)");

    // 4. 将 \_ 替换为普通下划线（移除转义）
    std::regex underscore(R"(\\_)");
    result = std::regex_replace(result, underscore, "_");

    // 5. 将 \text{null} 替换为 null
    std::regex textNull(R"(\\text\{null\})");
    result = std::regex_replace(result, textNull, "null");

    // 6. 移除控制结构前的 \State（\State \If -> \If 等）
    //    同样适用于 \While、\For、\ElsIf、\Else
    std::regex stateControl(R"(\\State\s+(\\If|\\While|\\For|\\ElsIf|\\Else))");
    result = std::regex_replace(result, stateControl, "$1");

    // 7. 将单独的 \Return 临时转换为 \State \Return（便于统一处理）
    std::regex returnPattern(R"((?<!\\)\\Return(?!\w))");
    result = std::regex_replace(result, returnPattern, "\\State \\Return");

    // 8. 去除多余空行（连续换行合并为一个）
    std::regex blankLines(R"(\n\s*\n+)");
    result = std::regex_replace(result, blankLines, "\n");

    // 9. 将 \State \Return 转换回 \Return（最终统一）
    std::regex stateReturn(R"(\\State\s+\\Return)");
    result = std::regex_replace(result, stateReturn, "\\Return");

    return result;
}

/**
 * 从整个文件中提取并转换所有 algorithmic 环境
 */
std::string convertFile(const std::string& content) {
    // 匹配所有 \begin{algorithmic}...\end{algorithmic} 块
    // 使用非贪婪匹配以支持多个块
    std::regex algorithmicBlock(R"(\\begin\{algorithmic\}(.*?)\\end\{algorithmic\})",
                                std::regex::ECMAScript | std::regex::optimize);

    std::sregex_iterator it(content.begin(), content.end(), algorithmicBlock);
    std::sregex_iterator end;

    std::stringstream output;
    bool first = true;

    for (; it != end; ++it) {
        std::smatch match = *it;
        std::string inner = match[1].str();        // 内部内容（不含开始/结束标签）
        std::string transformed = convertAlgorithmicContent(inner);

        if (!first) output << "\n\n";
        first = false;

        // 重新组装成完整的 algorithmic 环境
        output << "\\begin{algorithmic}\n" << transformed << "\n\\end{algorithmic}";
    }

    // 如果没有找到 algorithmic 块，则对整个内容进行基本清理
    if (first) {
        // 可能整个内容就是 algorithmic 体（无外层标签），我们仍尝试转换
        // 但为了安全，我们只转换常见的错误，并保留原样
        std::string cleaned = convertAlgorithmicContent(content);
        return cleaned;
    }

    return output.str();
}

int main(int argc, char* argv[]) {
    if (argc < 2) {
        std::cerr << "用法: " << argv[0] << " <输入文件> [输出文件]\n";
        std::cerr << "  若未指定输出文件，结果打印到标准输出。\n";
        return 1;
    }

    std::string inputFile = argv[1];
    std::ifstream inFile(inputFile);
    if (!inFile.is_open()) {
        std::cerr << "错误：无法打开输入文件 " << inputFile << "\n";
        return 1;
    }

    std::stringstream buffer;
    buffer << inFile.rdbuf();
    std::string content = buffer.str();
    inFile.close();

    std::string converted = convertFile(content);

    if (argc >= 3) {
        std::string outputFile = argv[2];
        std::ofstream outFile(outputFile);
        if (!outFile.is_open()) {
            std::cerr << "错误：无法创建输出文件 " << outputFile << "\n";
            return 1;
        }
        outFile << converted;
        outFile.close();
        std::cout << "转换成功，结果已写入 " << outputFile << "\n";
    } else {
        std::cout << converted;
    }

    return 0;
}
```

```C
/**
 * main.c
 *
 * 将 LaTeX algorithmic 伪代码转换为 Pseudocode.js 可识别的格式。
 * 编译: gcc -std=c99 -O2 main.c -o main
 * 用法: ./main <输入文件> [输出文件]
 *
 * 功能：
 *   - 自动提取所有 \begin{algorithmic}...\end{algorithmic} 块
 *   - 去除外层 \begin{algorithm}...\end{algorithm}
 *   - 修复拼写错误：\CommentRR → \Comment
 *   - 删除无效命令：\quad
 *   - 转换 \Call{func}{arg} → func(arg)（简单形式）
 *   - 移除 \_ 转义（\_right → _right）
 *   - 替换 \text{null} → null
 *   - 移除控制结构前的 \State（\State \If → \If）
 *   - 将 \State \Return 统一转换为 \Return
 *   - 合并多余空行
 */

#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include <ctype.h>

/* ---------- 动态字符串辅助 ---------- */
typedef struct {
    char *data;
    size_t len;
    size_t cap;
} String;

void str_init(String *s) {
    s->data = NULL;
    s->len = 0;
    s->cap = 0;
}

void str_free(String *s) {
    free(s->data);
    s->data = NULL;
    s->len = s->cap = 0;
}

void str_append_char(String *s, char c) {
    if (s->len + 1 >= s->cap) {
        size_t new_cap = s->cap ? s->cap * 2 : 16;
        char *new_data = realloc(s->data, new_cap);
        if (!new_data) { fprintf(stderr, "内存分配失败\n"); exit(1); }
        s->data = new_data;
        s->cap = new_cap;
    }
    s->data[s->len++] = c;
    s->data[s->len] = '\0';
}

void str_append_str(String *s, const char *str) {
    while (*str) str_append_char(s, *str++);
}

void str_append_n(String *s, const char *str, size_t n) {
    for (size_t i = 0; i < n; ++i) str_append_char(s, str[i]);
}

void str_clear(String *s) {
    s->len = 0;
    if (s->data) s->data[0] = '\0';
}

/* ---------- 字符串替换/删除工具 ---------- */
/* 将 src 中所有出现的 old 替换为 new，返回新字符串（调用者释放） */
char *str_replace_all(const char *src, const char *old, const char *new) {
    String result;
    str_init(&result);
    const char *p = src;
    size_t old_len = strlen(old);
    size_t new_len = strlen(new);
    while (*p) {
        if (strncmp(p, old, old_len) == 0) {
            str_append_str(&result, new);
            p += old_len;
        } else {
            str_append_char(&result, *p++);
        }
    }
    return result.data;  // 调用者负责 free
}

/* 删除 src 中所有出现的 old，返回新字符串 */
char *str_remove_all(const char *src, const char *old) {
    return str_replace_all(src, old, "");
}

/* 去除字符串前后的空白（包括换行） */
char *str_trim(char *s) {
    while (isspace((unsigned char)*s)) ++s;
    if (*s == '\0') return s;
    char *end = s + strlen(s) - 1;
    while (end > s && isspace((unsigned char)*end)) --end;
    *(end + 1) = '\0';
    return s;
}

/* ---------- 核心转换函数 ---------- */
/*
 * 处理单个 algorithmic 块内部内容
 * 返回新分配的字符串，调用者负责释放
 */
char *convert_algorithmic_content(const char *input) {
    char *result = strdup(input);
    if (!result) { perror("strdup"); exit(1); }

    // 1. 替换 \CommentRR -> \Comment
    char *tmp = str_replace_all(result, "\\CommentRR", "\\Comment");
    free(result);
    result = tmp;

    // 2. 删除所有 \quad
    tmp = str_remove_all(result, "\\quad");
    free(result);
    result = tmp;

    // 3. 处理 \Call{func}{arg} -> func(arg)
    //    简单扫描，假设 func 和 arg 都不含花括号
    String converted;
    str_init(&converted);
    const char *p = result;
    while (*p) {
        if (strncmp(p, "\\Call{", 6) == 0) {
            const char *start_func = p + 6;
            const char *end_func = strchr(start_func, '}');
            if (!end_func) { /* 格式错误，直接复制 */ str_append_char(&converted, *p++); continue; }
            const char *start_arg = end_func + 1;
            if (*start_arg != '{') { /* 格式错误 */ str_append_char(&converted, *p++); continue; }
            const char *end_arg = strchr(start_arg + 1, '}');
            if (!end_arg) { /* 格式错误 */ str_append_char(&converted, *p++); continue; }
            // 提取函数名和参数
            size_t func_len = end_func - start_func;
            size_t arg_len = end_arg - (start_arg + 1);
            str_append_n(&converted, start_func, func_len);
            str_append_char(&converted, '(');
            str_append_n(&converted, start_arg + 1, arg_len);
            str_append_char(&converted, ')');
            p = end_arg + 1;  // 跳过整个 \Call{...}{...}
            continue;
        }
        str_append_char(&converted, *p++);
    }
    free(result);
    result = converted.data;

    // 4. 将 \_ 替换为 _（移除转义）
    tmp = str_replace_all(result, "\\_", "_");
    free(result);
    result = tmp;

    // 5. 将 \text{null} 替换为 null
    tmp = str_replace_all(result, "\\text{null}", "null");
    free(result);
    result = tmp;

    // 6. 移除控制结构前的 \State: \State \If -> \If 等
    //    处理 \State \If, \State \While, \State \For, \State \ElsIf, \State \Else
    const char *patterns[] = {"\\State \\If", "\\State \\While", "\\State \\For",
                              "\\State \\ElsIf", "\\State \\Else", NULL};
    for (int i = 0; patterns[i]; ++i) {
        tmp = str_replace_all(result, patterns[i], patterns[i] + 6); // 去掉 "\State "
        free(result);
        result = tmp;
    }

    // 7. 统一 \Return 处理：先将单独的 \Return 转为 \State \Return，再统一转回
    //    这样可以处理各种情况
    tmp = str_replace_all(result, "\\Return", "\\State \\Return");
    free(result);
    result = tmp;
    // 再转回 \Return（去掉 \State）
    tmp = str_replace_all(result, "\\State \\Return", "\\Return");
    free(result);
    result = tmp;

    // 8. 去除多余空行（连续换行合并为一个）
    char *trimmed = result;
    String clean;
    str_init(&clean);
    int last_was_newline = 0;
    while (*trimmed) {
        if (*trimmed == '\n') {
            if (!last_was_newline) {
                str_append_char(&clean, '\n');
                last_was_newline = 1;
            }
        } else {
            str_append_char(&clean, *trimmed);
            last_was_newline = 0;
        }
        ++trimmed;
    }
    free(result);
    result = clean.data;

    // 去掉首尾空白（但保留内部结构）
    char *trim_start = result;
    while (isspace((unsigned char)*trim_start)) ++trim_start;
    if (trim_start != result) {
        memmove(result, trim_start, strlen(trim_start) + 1);
    }
    // 去掉末尾空白（保留一个换行）
    char *end = result + strlen(result) - 1;
    while (end > result && isspace((unsigned char)*end)) --end;
    *(end + 1) = '\0';

    return result;
}

/*
 * 从整个文件中提取所有 algorithmic 块并转换
 * 返回新字符串（所有转换后的块连接，每个块用 \begin{algorithmic}...\end{algorithmic} 包裹）
 */
char *convert_file(const char *content) {
    String output;
    str_init(&output);

    const char *p = content;
    int first = 1;

    while (1) {
        const char *begin = strstr(p, "\\begin{algorithmic}");
        if (!begin) break;
        const char *end = strstr(begin + 18, "\\end{algorithmic}");
        if (!end) {
            fprintf(stderr, "错误: 未找到匹配的 \\end{algorithmic}\n");
            break;
        }

        // 提取内部内容（跳过 \begin{algorithmic} 和 \end{algorithmic}）
        const char *inner_start = begin + 18; // strlen("\\begin{algorithmic}") = 18
        size_t inner_len = end - inner_start;

        // 复制内部内容
        char *inner = malloc(inner_len + 1);
        if (!inner) { perror("malloc"); exit(1); }
        strncpy(inner, inner_start, inner_len);
        inner[inner_len] = '\0';

        // 转换内容
        char *converted = convert_algorithmic_content(inner);
        free(inner);

        // 组装块
        if (!first) str_append_str(&output, "\n\n");
        first = 0;
        str_append_str(&output, "\\begin{algorithmic}\n");
        str_append_str(&output, converted);
        str_append_str(&output, "\n\\end{algorithmic}");

        free(converted);
        p = end + 16; // 跳过 \end{algorithmic}
    }

    // 如果没有找到任何块，则对整个内容做基本清理（可能没有标签）
    if (first) {
        char *cleaned = convert_algorithmic_content(content);
        str_append_str(&output, cleaned);
        free(cleaned);
    }

    return output.data;
}

/* ---------- 主程序 ---------- */
int main(int argc, char *argv[]) {
    if (argc < 2) {
        fprintf(stderr, "用法: %s <输入文件> [输出文件]\n", argv[0]);
        fprintf(stderr, "  若未指定输出文件，结果打印到标准输出。\n");
        return 1;
    }

    FILE *in = fopen(argv[1], "rb");
    if (!in) {
        perror("无法打开输入文件");
        return 1;
    }

    fseek(in, 0, SEEK_END);
    long file_size = ftell(in);
    fseek(in, 0, SEEK_SET);
    char *buffer = malloc(file_size + 1);
    if (!buffer) { perror("内存分配失败"); fclose(in); return 1; }
    size_t read_size = fread(buffer, 1, file_size, in);
    buffer[read_size] = '\0';
    fclose(in);

    char *converted = convert_file(buffer);
    free(buffer);

    if (argc >= 3) {
        FILE *out = fopen(argv[2], "w");
        if (!out) {
            perror("无法创建输出文件");
            free(converted);
            return 1;
        }
        fputs(converted, out);
        fclose(out);
        printf("转换成功，结果已写入 %s\n", argv[2]);
    } else {
        puts(converted);
    }

    free(converted);
    return 0;
}
```

```Python
#!/usr/bin/env python3
# -*- coding: utf-8 -*-

"""
main.py

将 LaTeX algorithmic 伪代码转换为 Pseudocode.js 可识别的格式。

用法:
    python main.py <输入文件> [输出文件]
    若不指定输出文件，结果打印到标准输出。

功能:
    - 自动提取所有 \begin{algorithmic}...\end{algorithmic} 块
    - 去除外层 \begin{algorithm}...\end{algorithm}
    - 修复拼写错误：\CommentRR → \Comment
    - 删除无效命令：\quad
    - 转换 \Call{func}{arg} → func(arg)（简单形式）
    - 移除 \_ 转义（\_right → _right）
    - 替换 \text{null} → null
    - 移除控制结构前的 \State（\State \If → \If）
    - 将 \State \Return 统一转换为 \Return
    - 合并多余空行
"""

import re
import sys

def convert_algorithmic_content(text):
    """
    处理单个 algorithmic 块内部内容，返回转换后的字符串
    """
    # 1. 修复拼写错误 \CommentRR -> \Comment
    text = re.sub(r'\\CommentRR', r'\\Comment', text)

    # 2. 删除所有 \quad
    text = re.sub(r'\\quad\s*', '', text)

    # 3. 将 \Call{func}{arg} 转换为 func(arg)
    #    匹配 \Call{...}{...}，参数不能包含嵌套花括号（简化）
    def replace_call(match):
        func = match.group(1).strip()
        arg = match.group(2).strip()
        return f"{func}({arg})"
    text = re.sub(r'\\Call\{([^}]*)\}\{([^}]*)\}', replace_call, text)

    # 4. 将 \_ 替换为普通下划线（移除转义）
    text = re.sub(r'\\_', '_', text)

    # 5. 将 \text{null} 替换为 null
    text = re.sub(r'\\text\{null\}', 'null', text)

    # 6. 移除控制结构前的 \State: \State \If -> \If, \State \While -> \While, etc.
    text = re.sub(r'\\State\s+(\\If|\\While|\\For|\\ElsIf|\\Else)', r'\1', text)

    # 7. 统一 \Return：先将所有 \Return 变成 \State \Return，再统一转回 \Return
    text = re.sub(r'\\Return', r'\\State \\Return', text)
    text = re.sub(r'\\State\s+\\Return', r'\\Return', text)

    # 8. 去除多余空行（连续换行合并为一个）
    text = re.sub(r'\n\s*\n+', '\n', text)

    # 去掉首尾空白（保留内部结构）
    text = text.strip()
    # 确保末尾至少有一个换行（为了美观）
    if text and text[-1] != '\n':
        text += '\n'

    return text

def convert_file(content):
    """
    从整个文件中提取所有 algorithmic 块并转换，返回完整的转换后内容
    """
    # 匹配所有 \begin{algorithmic}...\end{algorithmic} 块（非贪婪）
    pattern = re.compile(r'\\begin\{algorithmic\}(.*?)\\end\{algorithmic\}', re.DOTALL)
    matches = pattern.findall(content)

    if not matches:
        # 如果没有找到块，对整个内容做基本清理（可能没有标签）
        cleaned = convert_algorithmic_content(content)
        return cleaned

    output_parts = []
    for inner in matches:
        converted = convert_algorithmic_content(inner)
        block = f"\\begin{{algorithmic}}\n{converted}\\end{{algorithmic}}"
        output_parts.append(block)

    return '\n\n'.join(output_parts)

def main():
    if len(sys.argv) < 2:
        print(f"用法: {sys.argv[0]} <输入文件> [输出文件]")
        print("  若未指定输出文件，结果打印到标准输出。")
        sys.exit(1)

    input_file = sys.argv[1]
    try:
        with open(input_file, 'r', encoding='utf-8') as f:
            content = f.read()
    except FileNotFoundError:
        print(f"错误：文件 '{input_file}' 未找到。", file=sys.stderr)
        sys.exit(1)
    except Exception as e:
        print(f"读取文件时出错：{e}", file=sys.stderr)
        sys.exit(1)

    converted = convert_file(content)

    if len(sys.argv) >= 3:
        output_file = sys.argv[2]
        try:
            with open(output_file, 'w', encoding='utf-8') as f:
                f.write(converted)
            print(f"转换成功，结果已写入 {output_file}")
        except Exception as e:
            print(f"写入文件时出错：{e}", file=sys.stderr)
            sys.exit(1)
    else:
        print(converted)

if __name__ == "__main__":
    main()
```

```java
import java.io.*;
import java.nio.charset.StandardCharsets;
import java.nio.file.*;
import java.util.regex.*;

/**
 * ConvertPseudocode.java
 * 
 * 将 LaTeX algorithmic 伪代码转换为 Pseudocode.js 可识别的格式。
 * 
 * 编译: javac ConvertPseudocode.java
 * 用法: java ConvertPseudocode <输入文件> [输出文件]
 *       若不指定输出文件，结果打印到标准输出。

 *   - 自动提取所有 \begin{algorithmic}...\end{algorithmic} 块
 *   - 去除外层 \begin{algorithm}...\end{algorithm}
 *   - 修复拼写错误：\CommentRR → \Comment
 *   - 删除无效命令：\quad
 *   - 转换 \Call{func}{arg} → func(arg)（简单形式）
 *   - 移除 \_ 转义（\_right → _right）
 *   - 替换 \text{null} → null
 *   - 移除控制结构前的 \State（\State \If → \If）
 *   - 将 \State \Return 统一转换为 \Return
 *   - 合并多余空行
 */
public class ConvertPseudocode {

    /**
     * 处理单个 algorithmic 块内部内容，返回转换后的字符串
     */
    private static String convertAlgorithmicContent(String text) {
        // 1. 修复拼写错误 \CommentRR -> \Comment
        text = text.replaceAll("\\\\CommentRR", "\\\\Comment");

        // 2. 删除所有 \quad（及其后的空白）
        text = text.replaceAll("\\\\quad\\s*", "");

        // 3. 将 \Call{func}{arg} 转换为 func(arg)
        //    匹配 \Call{...}{...}，参数不能包含嵌套花括号（简化）
        Pattern callPattern = Pattern.compile("\\\\Call\\{([^}]*)\\}\\{([^}]*)\\}");
        Matcher callMatcher = callPattern.matcher(text);
        StringBuffer sb = new StringBuffer();
        while (callMatcher.find()) {
            String func = callMatcher.group(1).trim();
            String arg = callMatcher.group(2).trim();
            callMatcher.appendReplacement(sb, func + "(" + arg + ")");
        }
        callMatcher.appendTail(sb);
        text = sb.toString();

        // 4. 将 \_ 替换为普通下划线（移除转义）
        text = text.replaceAll("\\\\_", "_");

        // 5. 将 \text{null} 替换为 null
        text = text.replaceAll("\\\\text\\{null\\}", "null");

        // 6. 移除控制结构前的 \State: \State \If -> \If, \State \While -> \While, etc.
        text = text.replaceAll("\\\\State\\s+(\\\\If|\\\\While|\\\\For|\\\\ElsIf|\\\\Else)", "$1");

        // 7. 统一 \Return：先将所有 \Return 变成 \State \Return，再统一转回 \Return
        text = text.replaceAll("\\\\Return", "\\\\State \\\\Return");
        text = text.replaceAll("\\\\State\\s+\\\\Return", "\\\\Return");

        // 8. 去除多余空行（连续换行合并为一个）
        text = text.replaceAll("\\n\\s*\\n+", "\n");

        // 去掉首尾空白（保留内部结构）
        text = text.trim();
        // 确保末尾至少有一个换行（为了美观）
        if (!text.isEmpty() && text.charAt(text.length() - 1) != '\n') {
            text += "\n";
        }

        return text;
    }

    /**
     * 从整个文件中提取所有 algorithmic 块并转换，返回完整的转换后内容
     */
    private static String convertFile(String content) {
        // 匹配所有 \begin{algorithmic}...\end{algorithmic} 块（非贪婪）
        Pattern blockPattern = Pattern.compile(
                "\\\\begin\\{algorithmic\\}(.*?)\\\\end\\{algorithmic\\}",
                Pattern.DOTALL
        );
        Matcher blockMatcher = blockPattern.matcher(content);

        // 如果找不到任何块，则对整个内容做基本清理
        if (!blockMatcher.find()) {
            return convertAlgorithmicContent(content);
        }

        // 重置匹配器，重新遍历
        blockMatcher.reset();
        StringBuilder result = new StringBuilder();
        boolean first = true;

        while (blockMatcher.find()) {
            String inner = blockMatcher.group(1);
            String converted = convertAlgorithmicContent(inner);

            if (!first) {
                result.append("\n\n");
            }
            first = false;
            result.append("\\begin{algorithmic}\n")
                  .append(converted)
                  .append("\\end{algorithmic}");
        }

        return result.toString();
    }

    public static void main(String[] args) {
        if (args.length < 1) {
            System.err.println("用法: java ConvertPseudocode <输入文件> [输出文件]");
            System.err.println("  若未指定输出文件，结果打印到标准输出。");
            System.exit(1);
        }

        String inputFile = args[0];
        String content;
        try {
            content = new String(Files.readAllBytes(Paths.get(inputFile)), StandardCharsets.UTF_8);
        } catch (IOException e) {
            System.err.println("读取文件时出错: " + e.getMessage());
            System.exit(1);
            return;
        }

        String converted = convertFile(content);

        if (args.length >= 2) {
            String outputFile = args[1];
            try {
                Files.write(Paths.get(outputFile), converted.getBytes(StandardCharsets.UTF_8));
                System.out.println("转换成功，结果已写入 " + outputFile);
            } catch (IOException e) {
                System.err.println("写入文件时出错: " + e.getMessage());
                System.exit(1);
            }
        } else {
            System.out.print(converted);
        }
    }
}
```

### Shell

```bash
#!/bin/bash
# main.sh
# 用法: ./main.sh <输入文件> [输出文件]
# 若不指定输出文件，结果打印到标准输出。

# 检查参数
if [ $# -lt 1 ]; then
    echo "用法: $0 <输入文件> [输出文件]"
    exit 1
fi

input="$1"
output="$2"

# 检查是否安装了 perl
if ! command -v perl &> /dev/null; then
    echo "错误: 需要 perl 但未安装。"
    exit 1
fi

# 读取整个文件
content=$(cat "$input")

# 定义转换函数（处理单个 algorithmic 块内部内容）
convert_block() {
    local block="$1"
    # 1. 替换 \CommentRR -> \Comment
    block=$(echo "$block" | sed 's/\\CommentRR/\\Comment/g')
    # 2. 删除 \quad 及其后的空白
    block=$(echo "$block" | sed 's/\\quad[[:space:]]*//g')
    # 3. 处理 \Call{func}{arg} -> func(arg)  使用 Perl 正则（非贪婪）
    block=$(echo "$block" | perl -pe 's/\\Call\{([^}]*)\}\{([^}]*)\}/$1($2)/g')
    # 4. 将 \_ 替换为 _
    block=$(echo "$block" | sed 's/\\_/_/g')
    # 5. 将 \text{null} 替换为 null
    block=$(echo "$block" | sed 's/\\text{null}/null/g')
    # 6. 移除控制结构前的 \State: \State \If -> \If 等
    block=$(echo "$block" | sed -E 's/\\State[[:space:]]+(\\If|\\While|\\For|\\ElsIf|\\Else)/\1/g')
    # 7. 统一 \Return: 先转成 \State \Return，再统一转回
    block=$(echo "$block" | sed 's/\\Return/\\State \\Return/g')
    block=$(echo "$block" | sed 's/\\State[[:space:]]*\\Return/\\Return/g')
    # 8. 压缩多余空行
    block=$(echo "$block" | sed '/^[[:space:]]*$/d' | sed '$!G' | sed -e 's/^\n//' -e 's/\n$//')
    # 如果最后没有换行，补一个
    if [ -n "$block" ] && [ "${block: -1}" != $'\n' ]; then
        block="$block"$'\n'
    fi
    echo "$block"
}

# 检查是否包含 \begin{algorithmic}...\end{algorithmic} 块
if echo "$content" | grep -q '\\begin{algorithmic}'; then
    # 提取多个块并分别处理
    # 使用 Perl 提取块，这里用简单方法：使用 awk 或 perl 分段
    # 更稳健：用 Perl 一次性处理
    result=$(echo "$content" | perl -0777 -pe '
        s/\\begin\{algorithmic\}(.*?)\\end\{algorithmic\}/ 
            $inner = $1;
            # 应用转换
            $inner =~ s/\\\\CommentRR/\\\\Comment/g;
            $inner =~ s/\\\\quad\s*//g;
            $inner =~ s/\\\\Call\{([^}]*)\}\{([^}]*)\}/$1($2)/g;
            $inner =~ s/\\\\_/_/g;
            $inner =~ s/\\\\text\{null\}/null/g;
            $inner =~ s/\\\\State\s+(\\\\If|\\\\While|\\\\For|\\\\ElsIf|\\\\Else)/$1/g;
            $inner =~ s/\\\\Return/\\\\State \\\\Return/g;
            $inner =~ s/\\\\State\s+\\\\Return/\\\\Return/g;
            $inner =~ s/\n\s*\n+/\n/g;
            $inner =~ s/^\s+//; $inner =~ s/\s+$//;
            $inner .= "\n" if $inner !~ /\n$/;
            "\\\\begin{algorithmic}\n$inner\\\\end{algorithmic}"
        /ge;
        print;
    ')
else
    # 无块，对整个文件做基本清理
    result=$(echo "$content" | perl -pe '
        s/\\\\CommentRR/\\\\Comment/g;
        s/\\\\quad\s*//g;
        s/\\\\Call\{([^}]*)\}\{([^}]*)\}/$1($2)/g;
        s/\\\\_/_/g;
        s/\\\\text\{null\}/null/g;
        s/\\\\State\s+(\\\\If|\\\\While|\\\\For|\\\\ElsIf|\\\\Else)/$1/g;
        s/\\\\Return/\\\\State \\\\Return/g;
        s/\\\\State\s+\\\\Return/\\\\Return/g;
        s/\n\s*\n+/\n/g;
    ')
fi

# 输出结果
if [ -n "$output" ]; then
    echo "$result" > "$output"
    echo "转换成功，结果已写入 $output"
else
    echo "$result"
fi
```
